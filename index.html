☁️ เชื่อมระบบคลาวด์ Google Sheets

ทำให้มือถือและคอมพิวเตอร์ทุกเครื่องเปิดดูข้อมูลตรงกันและลบพร้อมกันได้ทันที

สถานะของระบบ: จัดเก็บแบบส่วนตัวในเครื่องนี้ (ออฟไลน์)

💡 วิธีติดตั้งให้ลิงก์กันหมดทุกเครื่อง (ฟรีใน 5 นาที):

  1. เปิด Google Sheets ใหม่ แล้วพิมพ์หัวตารางแถวที่ 1 คอลัมน์ A ถึง G ตามนี้เป๊ะ ๆ:
    id | name | category | lot | quantity | expiryDate | addedAt
  2. กดไปที่เมนู ส่วนขยาย (Extensions) > Apps Script
  3. ลบโค้ดเริ่มต้นออกให้เกลี้ยง แล้วนำโค้ดด้านล่างนี้ไปวางแทน:
function doGet(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = sheet.getDataRange().getValues();
  var headers = data[0];
  var result = [];
  for (var i = 1; i < data.length; i++) {
    var row = {};
    var hasData = false;
    for (var j = 0; j < headers.length; j++) {
      row[headers[j]] = data[i][j];
      if (data[i][j] !== "") hasData = true;
    }
    if (hasData) result.push(row);
  }
  return ContentService.createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON)
    .setHeader("Access-Control-Allow-Origin", "*");
}

function doPost(e) {
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    var param = JSON.parse(e.postData.contents);
    
    if (param.action === "add") {
      sheet.appendRow([
        param.id,
        param.name,
        param.category || '',
        param.lot || '',
        param.quantity || 1,
        param.expiryDate,
        new Date().toLocaleString('th-TH')
      ]);
      return ContentService.createTextOutput(JSON.stringify({status: "success"}))
        .setMimeType(ContentService.MimeType.JSON)
        .setHeader("Access-Control-Allow-Origin", "*");
    } 
    
    else if (param.action === "delete") {
      var data = sheet.getDataRange().getValues();
      var idColumnIndex = 0; // คอลัมน์ A คือ ID
      var rowDeleted = false;
      
      for (var i = data.length - 1; i >= 1; i--) {
        if (data[i][idColumnIndex] == param.id) {
          sheet.deleteRow(i + 1);
          rowDeleted = true;
        }
      }
      
      if (rowDeleted) {
        return ContentService.createTextOutput(JSON.stringify({status: "success"}))
          .setMimeType(ContentService.MimeType.JSON)
          .setHeader("Access-Control-Allow-Origin", "*");
      } else {
        return ContentService.createTextOutput(JSON.stringify({status: "error", message: "ไม่พบสินค้า"}))
          .setMimeType(ContentService.MimeType.JSON)
          .setHeader("Access-Control-Allow-Origin", "*");
      }
    }
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({status: "error", message: err.message}))
      .setMimeType(ContentService.MimeType.JSON)
      .setHeader("Access-Control-Allow-Origin", "*");
  }
}

4. กดปุ่ม การทำให้ใช้งานได้ (Deploy) > การทำให้ใช้งานได้ใหม่ (New Deployment)
- เลือกประเภทเป็น แอปพลิเคชันเว็บ (Web App)
- ตั้งค่าผู้ที่มีสิทธิ์เข้าถึงเป็น ทุกคน (Anyone)
- กดปุ่ม "ทำให้ใช้งานได้" จากนั้นอนุญาตสิทธิ์ แล้วนำ URL ที่ได้มาแปะในระบบนี้ได้เลยครับ!
